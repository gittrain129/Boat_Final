--현재 진행중 테이블입니다.230329
create table TODOLIST (
	NUM VARCHAR2(12),
	EMPNO VARCHAR2(12) references member(empno),
	T_CONTENT VARCHAR2(200),
	START_DATE VARCHAR2(15),
	END_DATE VARCHAR2(15),
	REG_DATE  DATE DEFAULT SYSDATE,	
	DEPT VARCHAR2(15)
);



--부서별 Todo값 가져오기
SELECT *
FROM (select * from todolist where dept = '인사팀')T
 JOIN (select * from member where dept = '인사팀')M
ON T.empno = M.empno
order by empno;
