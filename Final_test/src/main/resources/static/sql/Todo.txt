drop table TODOLIST cascade constraints purge;

create sequence todo_seq nocache;

create table TODOLIST (
	NUM VARCHAR2(12),
	EMPNO VARCHAR2(1000),
	T_CONTENT VARCHAR2(200),
	START_DATE VARCHAR2(15),
	END_DATE VARCHAR2(15),
	REG_DATE  DATE DEFAULT SYSDATE,	
	DEPT VARCHAR2(15)
);



COL NUM FORMAT A20
COL F_C_ID FORMAT A20
COL EMPNO FORMAT A20
COL T_CONTENT FORMAT A20
COL START_DATE FORMAT A20
COL END_DATE FORMAT A20
COL DEPT FORMAT A20
COL REG_DATE FORMAT A20
